From src/aquapose/core/association/scoring.py (scalar reference):
```python
def ray_ray_closest_point(
    origin_a: np.ndarray,   # (3,)
    dir_a: np.ndarray,      # (3,)
    origin_b: np.ndarray,   # (3,)
    dir_b: np.ndarray,      # (3,)
) -> tuple[float, np.ndarray]:
    """Returns (distance, midpoint). Midpoint shape (3,)."""
    w0 = origin_a - origin_b
    a = float(np.dot(dir_a, dir_a))
    b = float(np.dot(dir_a, dir_b))
    c = float(np.dot(dir_b, dir_b))
    d = float(np.dot(dir_a, w0))
    e = float(np.dot(dir_b, w0))
    denom = a * c - b * b
    # Near-parallel guard: abs(denom) < 1e-12
    # fallback: t_a=0, s_b = e/c if abs(c)>1e-12 else 0.0
    t_a = (b * e - c * d) / denom
    s_b = (a * e - b * d) / denom
    pt_a = origin_a + t_a * dir_a
    pt_b = origin_b + s_b * dir_b
    dist = float(np.linalg.norm(pt_a - pt_b))
```
